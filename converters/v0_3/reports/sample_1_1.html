import json
from pathlib import Path

import numpy as np
import plotly.graph_objects as go

# ── Load theme from theme/font_mapping.json ──────────────────────
# Walks up from this script's directory until it finds a "theme" folder.
THEME_NAME = "navy_vault"

def _find_theme_file() -> Path:
    try:
        anchor = Path(__file__).resolve().parent
    except NameError:
        anchor = Path.cwd()  # fallback for Jupyter notebooks
    for parent in [anchor, *anchor.parents]:
        candidate = parent / "theme" / "font_mapping.json"
        if candidate.is_file():
            return candidate
    raise FileNotFoundError("Could not find theme/font_mapping.json")

_cfg = json.loads(_find_theme_file().read_text())
C    = _cfg["themes"][THEME_NAME]
FONT = _cfg["font"]["family"]
FSIZ = _cfg["font"]["size"]

# ── Data ─────────────────────────────────────────────────────────
x = np.linspace(0, 4 * np.pi, 500)

fig = go.Figure()
fig.add_trace(go.Scatter(
    x=x, y=np.sin(x),
    mode="lines",
    name="sin(x)",
    line=dict(color=C["mid"], width=2.5),
    fill="tozeroy",
    fillcolor="rgba(14,96,144,0.08)",
))

# ── Slider steps ─────────────────────────────────────────────────
freq_steps = []
for f in np.arange(0.5, 5.1, 0.5):
    freq_steps.append(dict(
        method="restyle",
        args=[{"y": [np.sin(f * x)]}],
        label=f"{f:.1f}",
    ))

amp_steps = []
for a in np.arange(0.5, 3.1, 0.5):
    amp_steps.append(dict(
        method="restyle",
        args=[{"y": [a * np.sin(x)]}],
        label=f"{a:.1f}",
    ))

phase_steps = []
for p in np.arange(0, 2 * np.pi + 0.01, np.pi / 6):
    phase_steps.append(dict(
        method="restyle",
        args=[{"y": [np.sin(x + p)]}],
        label=f"{p:.2f}",
    ))

# ── Slider styling (shared) ─────────────────────────────────────
slider_style = dict(
    bordercolor=C["silver"],
    bgcolor=C["ci"],
    activebgcolor=C["mid"],
    tickcolor=C["silver"],
    font=dict(family=FONT, size=FSIZ["sub"], color=C["subtext"]),
    ticklen=4,
    minorticklen=0,
    len=0.92,
    x=0.04,
)
_cv = dict(font=dict(family=FONT, size=FSIZ["label"], color=C["text"]))

fig.update_layout(
    sliders=[
        dict(
            **slider_style,
            active=1,
            currentvalue={**_cv, "prefix": "Frequency ω = "},
            steps=freq_steps,
            y=-0.02,
            pad=dict(t=30, b=10),
        ),
        dict(
            **slider_style,
            active=1,
            currentvalue={**_cv, "prefix": "Amplitude A = "},
            steps=amp_steps,
            y=-0.14,
            pad=dict(t=30, b=10),
        ),
        dict(
            **slider_style,
            active=0,
            currentvalue={**_cv, "prefix": "Phase φ = "},
            steps=phase_steps,
            y=-0.26,
            pad=dict(t=30, b=10),
        ),
    ],

    # ── Layout ───────────────────────────────────────────────────
    title=dict(
        text="y = A · sin(ω·x + φ)",
        font=dict(family=FONT, size=16, color=C["text"]),
        x=0.04, xanchor="left",
    ),
    paper_bgcolor=C["canvas"],
    plot_bgcolor=C["canvas"],
    font=dict(family=FONT, color=C["text"]),
    width=820,
    height=560,
    margin=dict(l=60, r=30, t=55, b=185),

    xaxis=dict(
        title=dict(text="x", font=dict(size=FSIZ["label"], color=C["subtext"])),
        tickfont=dict(size=FSIZ["tick"], color=C["subtext"]),
        gridcolor=C["grid"],
        zerolinecolor=C["silver"],
        zerolinewidth=0.8,
    ),
    yaxis=dict(
        title=dict(text="y", font=dict(size=FSIZ["label"], color=C["subtext"])),
        tickfont=dict(size=FSIZ["tick"], color=C["subtext"]),
        gridcolor=C["grid"],
        zerolinecolor=C["silver"],
        zerolinewidth=0.8,
    ),
    showlegend=False,
)

fig.show()
