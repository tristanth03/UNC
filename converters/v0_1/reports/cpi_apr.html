pltOverveiw <- function(tseries)
{
    par(mfrow = c(2,2))
    plot(tseries)
    qqnorm(tseries)
    qqline(tseries,col="red")
    acf(tseries)
    pacf(tseries)
}

search_modelSpace <- function(tseries,end)
{
    AR_p = vector(mode = "list",length = end)
    MA_q = vector(mode = "list",length = end)
    ARMA_pq = vector(mode = "list",length = end**2)

    labels_AR_p = vector(mode = "list",length = end)
    labels_MA_q = vector(mode = "list",length = end)
    labels_ARMA_pq = vector(mode = "list",length = end**2)

    AIC_AR_p = numeric(end)
    AIC_MA_q = numeric(end)
    AIC_ARMA_pq = numeric(end**2)

    for(i in 1:end)
    {
        AR_p[[i]] = arima(tseries,order = c(i,0,0),method = "ML")
        AIC_AR_p[i] = AR_p[[i]]$aic
        labels_AR_p[[i]] = paste0("AR(",i,")")
        MA_q[[i]] = arima(tseries,order = c(0,0,i),method = "ML")
        AIC_MA_q[i] = MA_q[[i]]$aic
        labels_MA_q[[i]] = paste0("MA(",i,")")

        for(j in 1:end)
        {
            k = j+(i-1)*end
        
            ARMA_pq[[k]] = arima(tseries,order = c(i,0,j),method = "ML") 
            labels_ARMA_pq[[k]] = paste0("ARMA(",i,",",j,")")
            AIC_ARMA_pq[k] = ARMA_pq[[k]]$aic
        }
    }

    plot(AIC_AR_p, type='p')
    points(AIC_MA_q, col='red')
    text(1:length(AIC_AR_p), AIC_AR_p, labels=unlist(labels_AR_p), pos=3, cex=0.7)
    text(1:length(AIC_MA_q), AIC_MA_q, labels=unlist(labels_MA_q), pos=1, cex=0.7)


    pq_sum <- sapply(1:length(ARMA_pq), function(k) {
        p <- ceiling(k / end)
        q <- ((k - 1) %% end) + 1
        p + q
    })

    pos_vec <- rep(c(1, 3), length.out = length(AIC_ARMA_pq))
    plot(pq_sum, AIC_ARMA_pq, col='red', xlab='p + q', ylab='AIC')
    text(pq_sum, AIC_ARMA_pq, labels=unlist(labels_ARMA_pq), pos=pos_vec, cex=0.7)


    return(list(
        AR_p = AR_p,
        MA_q = MA_q,
        ARMA_pq = ARMA_pq,
        labels_AR_p = labels_AR_p,
        labels_MA_q = labels_MA_q,
        labels_ARMA_pq = labels_ARMA_pq,
        AIC_AR_p = AIC_AR_p,
        AIC_MA_q = AIC_MA_q,
        AIC_ARMA_pq = AIC_ARMA_pq
    ))
}

pltOverveiw(res$ARMA_pq[[12]]$residuals)
model = res$ARMA_pq[[12]]
start_window <- 400
horizon <- 3

alpha95 <- 0.05
alpha99 <- 0.01
z95 <- qnorm(1 - alpha95 / 2)
z99 <- qnorm(1 - alpha99 / 2)

cpi.pred <- predict(model, n.ahead = horizon)

t_actual <- time(cpi)
n_total  <- length(t_actual)

t_start_plot <- t_actual[start_window]
t_forecast   <- seq(t_actual[n_total] + 1/12, by = 1/12, length.out = horizon)

last_cpi    <- as.numeric(cpi[n_total])
pred_undiff <- last_cpi + cumsum(cpi.pred$pred)
se_undiff   <- cpi.pred$se

x_forecast <- c(t_actual[n_total], t_forecast)
y_forecast  <- c(last_cpi, as.numeric(pred_undiff))
se_forecast <- c(0, as.numeric(se_undiff))

y_upper99 <- y_forecast + z99 * se_forecast
y_lower99 <- y_forecast - z99 * se_forecast
y_upper95 <- y_forecast + z95 * se_forecast
y_lower95 <- y_forecast - z95 * se_forecast

cpi_window <- window(cpi, start = t_start_plot)

     y    = as.numeric(cpi_window),
     type = "l",

     xlim = c(t_start_plot, max(t_forecast)),

     ylab = "CPI (%)",
     xlab = "Year",
     main = "CPI Forecast (Un-differenced)",
     xaxt = "n")

)
